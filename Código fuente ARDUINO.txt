void setup() {
  //MOTORES
  pinMode(6, OUTPUT); //Definimos los pines digitales como salidas
  pinMode(3, OUTPUT); 
  pinMode(9, OUTPUT); 
  pinMode(5, OUTPUT);

  //ULTRASONIDO
  Serial.begin(9600);
  pinMode(12, OUTPUT); /*activación del pin 9 como salida: para el pulso ultrasónico*/
  pinMode(10, INPUT);

  //LED
  pinMode(4, OUTPUT);   //Amarillo
  pinMode(7, OUTPUT);   //Rojo
  pinMode(2, OUTPUT);   //Verde

  //BLUETOOTH
  Serial.begin(9600);
}

void loop() { 
    //ESTABLECEMOS LA CONEXION
     if (Serial.available() > 0) 
    {string = "";}
    
    while(Serial.available() > 0)
    {
      command = ((byte)Serial.read());
      
      if(command == ':')
      {
        break;
      }
      
      else
      {
        string += command;
      }
      
      delay(1);
    }   

  digitalWrite(12,LOW); /* Por cuestión de estabilización del sensor*/

  delayMicroseconds(10);

if(string =="1" || string =="2" || string=="3" || string=="4" || string=="5" || string=="6")
